1. 버튼을 클릭해서 이미지를 화면에 출력해보자.


<div id="app1">
    <button v-on:click="showImg()">이미지 화면에 보이기</button>
    <div v-show="con"><img v-bind:src="imgSrc"></div>
</div>
<script>
   let app1 = new Vue({
       el: '#app1',
       data: {
           con: true,
           imgSrc: 'images/lion.jpg',
     },
       methods: {
          showImg: function () {
            this.con = !this.con;
           }
   }
   });
</script>
